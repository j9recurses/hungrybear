upstream app3_server {
  # point to app3 sock
  server unix:/var/www/hungrybear/tmp/sockets/unicorn_hungrybear.sock fail_timeout=0;
}


server {
  listen 80;
  server_name hungrybear.tk www.hungrybear.tk;
  root /var/www/hungrybear/public;

  location ^~ /assets/ {
    gzip_static on;
    expires max;
    add_header Cache-Control public;
  }

  try_files $uri @app3_server;
  location @app3_server {
  proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
  proxy_set_header Host $http_host;
  proxy_redirect off;
  proxy_pass http://app3_server;
 }

  error_page 500 502 503 504 /500.html;
  client_max_body_size 4G;
  keepalive_timeout 10;
}
